def rat_sem(x):
    if x == 'E' or x == 'K-A' or x == 'EC':
        return "Everyone" 
    elif x =='T':
        return "Teens"
    elif x == 'M':
        return "Mature"
    elif x == 'AO':
        return "Adults only"
    elif x == 'E10+':
        return "Everyone 10y+"
    elif x == "RP":
        return "Pending"
    else:
        return "Pending"
